### 6.2 CutoffCard Component

```tsx
interface CutoffCardProps {
  type: 'si' | 'vgm' | 'cargo' | 'gate' | 'doc';
  label: string;
  cutoffDate: string;
  status: 'submitted' | 'pending' | 'overdue';
  daysRemaining?: number;
  hoursRemaining?: number;
}

// Urgency calculation
// > 72h = safe (green border)
// 24-72h = warning (amber border)
// < 24h = urgent (red border, pulsing)
// past = overdue (red solid)
```
